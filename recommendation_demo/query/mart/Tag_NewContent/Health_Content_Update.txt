SELECT articles_id as article_id, tag, title as content, DATE(publish) as date FROM health_articles
WHERE categories_main != 37 AND (DATE(created_time) = CURRENT_DATE() OR DATE(created_time) = CURRENT_DATE() - INTERVAL 1 DAY OR DATE(created_time) = CURRENT_DATE() - INTERVAL 3 DAY OR DATE(created_time) = CURRENT_DATE() - INTERVAL 7 DAY OR DATE(created_time) = CURRENT_DATE() - INTERVAL 15 DAY OR DATE(created_time) = CURRENT_DATE() - INTERVAL 30 DAY OR DATE(created_time) = CURRENT_DATE() - INTERVAL 60 DAY OR DATE(created_time) = CURRENT_DATE() - INTERVAL 90 DAY OR DATE(created_time) = CURRENT_DATE() - INTERVAL 180 DAY OR DATE(created_time) = CURRENT_DATE() - INTERVAL 360 DAY OR DATE(created_time) = CURRENT_DATE() - INTERVAL 720 DAY)
AND articles_status = 1
